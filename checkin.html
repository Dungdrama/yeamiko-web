async function doCheckin() { // webUserId là ID tài khoản web (ví dụ lấy từ Firebase Auth) const webUserId = firebase.auth().currentUser.uid; const ref = database.ref('web_users/' + webUserId); const snap = await ref.once('value'); let d = snap.val() || { points: 0, count: 0, last: "" }; const today = new Date().toLocaleDateString(); if (d.last === today) return alert("Hôm nay bạn đã điểm danh rồi!"); await ref.update({ points: d.points + 2, count: (d.count >= 30) ? 1 : d.count + 1, last: today }); alert("Điểm danh thành công! +2 điểm."); }
